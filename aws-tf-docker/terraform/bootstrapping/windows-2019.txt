<powershell>
#
# This user-data content runs as 'Administrator'. The folder execution context is C:\Windows\System32
#
$VerbosePreference="Continue"
$ErrorActionPreference="Stop"

Set-Location $env:USERPROFILE

Set-Service -Name "ssh-agent" -StartupType Automatic
Start-Service ssh-agent

New-Item -Type Directory -Name ".ssh" -ErrorAction SilentlyContinue

"${private_key_content}" | Out-File -FilePath ~/.ssh/ssh-windows-to-linux.pem -Encoding ASCII

ssh-add .ssh/ssh-windows-to-linux.pem 

#
# Install Chocolatey...
#
Set-ExecutionPolicy Bypass -Scope Process -Force; [System.Net.ServicePointManager]::SecurityProtocol = [System.Net.ServicePointManager]::SecurityProtocol -bor 3072; iex ((New-Object System.Net.WebClient).DownloadString('https://chocolatey.org/install.ps1'))

#
# Install docker-cli
#
choco install docker-cli -y
</powershell>
