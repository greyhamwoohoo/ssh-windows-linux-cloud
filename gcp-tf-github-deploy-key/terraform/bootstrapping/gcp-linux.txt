# Startup Scripts are run as System every time the machine starts. 
# Reference: https://cloud.google.com/compute/docs/startupscript
#
# The Github Repo (github_repo_name) is cloned into /(github_repo_name) using the Deploy Key
#
set -ex

apt update -y

apt install git -y

#
# Clone repo using Github Deployment Key
#
echo "${github_deploy_private_key_content}" | base64 --decode > /tmp/github_deploy_private_key.pem
chmod 600 /tmp/github_deploy_private_key.pem

GIT_SSH_COMMAND="ssh -i /tmp/github_deploy_private_key.pem -o StrictHostKeyChecking=no" git clone git@github.com:${github_user_name}/${github_repo_name}

# chown -R ec2-user:ec2-user ${github_repo_name}
# cd ${github_repo_name}
# git remote remove origin 

rm /tmp/github_deploy_private_key.pem
